  def logit(x): return 1. / (1 + np.exp(-x))
  plt.figure(figsize=(4, 4))
  plt.plot(x, logit(x))
  plt.xlabel(r'$x$')
  plt.ylabel(r'$\mathrm{logit}$(x)')
  plt.title(r'Logit activation function')
  plt.tight_layout()
  fname = '/home/victor/org-roam/images/logit_activation.png'

  plt.close()
